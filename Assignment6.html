library(shiny)
Warning: package 'shiny' was built under R version 4.3.3
library(ggplot2)

# Define UI
ui <- fluidPage(
    titlePanel("Shiny App with Multiple Datasets and Iris Viewer"),
    
    sidebarLayout(
        sidebarPanel(
            # Dropdown for selecting the dataset
            selectInput("dataset", "Choose a dataset:", 
                        choices = c("mtcars", "USArrests", "uspop", "Iris Dataset")),
            
            uiOutput("varSelectUI")  # Dynamically generated UI for variable selection
        ),
        
        mainPanel(
            # Output for plot
            plotOutput("plot"),
            
            # Output for table
            tableOutput("dataTable")
        )
    )
)

# Define Server logic
server <- function(input, output) {
    
    # Reactive function to load the selected dataset
    datasetInput <- reactive({
        switch(input$dataset,
               "mtcars" = mtcars,
               "USArrests" = USArrests,
               "uspop" = data.frame(Year = as.numeric(time(uspop)), Population = as.numeric(uspop)),
               "Iris Dataset" = iris)
    })
    
    # Dynamically create a dropdown for variable selection based on the dataset
    output$varSelectUI <- renderUI({
        if (input$dataset == "mtcars") {
            selectInput("var", "Select variable:", choices = colnames(mtcars), selected = "mpg")
        } else if (input$dataset == "USArrests") {
            selectInput("var", "Select variable:", choices = colnames(USArrests), selected = "Murder")
        } else if (input$dataset == "Iris Dataset") {
            list(
                selectInput("xvar", "Select X Variable:", choices = names(iris)[1:4]),
                selectInput("yvar", "Select Y Variable:", choices = names(iris)[1:4]),
                selectInput("color", "Color by:", choices = names(iris)[5])
            )
        } else {
            NULL
        }
    })
    
    # Render the plot based on selected dataset and variables
    output$plot <- renderPlot({
        data <- datasetInput()
        
        if (input$dataset == "mtcars") {
            ggplot(data, aes(x = wt, y = .data[[input$var]])) + 
                geom_point() + 
                labs(title = "Mtcars Scatterplot", x = "Weight", y = input$var)
        } else if (input$dataset == "USArrests") {
            ggplot(data, aes(x = Murder, y = .data[[input$var]])) + 
                geom_point() + 
                labs(title = "US Arrests", x = "Murder", y = input$var)
        } else if (input$dataset == "uspop") {
            ggplot(data, aes(x = Year, y = Population)) + 
                geom_line(color = "blue") + 
                labs(title = "US Population Over Time", x = "Year", y = "Population")
        } else if (input$dataset == "Iris Dataset") {
            ggplot(data, aes_string(x = input$xvar, y = input$yvar, color = input$color)) + 
                geom_point(size = 3) + 
                labs(title = "Iris Dataset Scatter Plot") + 
                theme_minimal()
        }
    })
    
    # Render the table of selected dataset
    output$dataTable <- renderTable({
        datasetInput()
    })
}

# Run the Shiny app
shinyApp(ui = ui, server = server)
